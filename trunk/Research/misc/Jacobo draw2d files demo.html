<script type="text/javascript">
var workflow = new draw2d.Workflow("paintarea");
// Add an simple annotation to the canvas
//
var annotation = new draw2d.Annotation("NOTE: It is not binding that a connection ends in the corresponding port. See ConnectionAnchor for more information.");
annotation.setDimension(300,70);
workflow.addFigure(annotation,400,50);
var actor = new draw2d.shape.uml.Actor("Teacher");
workflow.addFigure(actor,150,150);
var object = new draw2d.shape.uml.Class("Object");
object.addAttribute("id","String");
workflow.addFigure(object,300,150);
var rectangle = new draw2d.shape.uml.Class("Rectangle");
rectangle.addAttribute("width","Integer","0");
rectangle.addAttribute("height","Integer","0");
workflow.addFigure(rectangle,100,300);
var circle = new draw2d.shape.uml.Class("Circle");
circle.addAttribute("diameter","Integer","0");
workflow.addFigure(circle,500,300);
var con = new draw2d.shape.uml.InheritanceConnection();
con.setSource(rectangle.portTop);
con.setTarget(object.portTop);
workflow.addFigure(con);
var con = new draw2d.shape.uml.InheritanceConnection();
con.setSource(circle.portTop);
con.setTarget(object.portTop);
workflow.addFigure(con);
</script>
